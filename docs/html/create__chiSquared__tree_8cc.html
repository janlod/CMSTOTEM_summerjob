= [mean, sigma](Int_t ntrk, Float_t branch, Float_t branchErr=0.0){
Float_t dev = 0.0;
Float_t value;
if(branchErr==0.0){
value = branch;
}else{
value = branch/branchErr;
}
if(ntrk==4){
dev = (mean - value)*(mean - value)/(sigma*sigma);
}
return dev;
}
